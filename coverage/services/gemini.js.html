async function withRateLimit(fn) {
  const now = Date.now()
  const wait = RATE_LIMIT_MS - (now - lastCallTime)
  if (wait > 0) await new Promise(r => setTimeout(r, wait))
  lastCallTime = Date.now()
  return fn()
}
